Почему код даёт именно такие результаты?

var a = 1, b = 1, c, d;

c = ++a; alert(c); // 2 a = 1 -> a = a + 1; c = a

d = b++; alert(d); // 1 b = 1 -> d = b; b = b + 1

c = (2+ ++a); alert(c); // 5 a = 2 -> a = a + 1; c = 2 + a

d = (2+ b++); alert(d); // 4 b = 2 -> d = 2 + b; b = b + 1

alert(a); // 3 ++a, ++a

alert(b); // 3 b++ b++
